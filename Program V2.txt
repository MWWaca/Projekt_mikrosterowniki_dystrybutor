void setup(){
  Serial.begin(9600);
  pinMode(IN_1, OUTPUT);
  pinMode(IN_2, OUTPUT);
  pinMode(IN_3, OUTPUT);
  pinMode(IN_4, OUTPUT);
  pinMode(sensorPin, INPUT);
  pinMode(buttonPinStop, INPUT_PULLUP);
  pinMode(buttonPinStart, INPUT_PULLUP);
  attachInterrupt(digitalPinToInterrupt(buttonPinStop), buttonStopISR, FALLING);
  attachInterrupt(digitalPinToInterrupt(buttonPinStart), buttonStartISR, CHANGE);
}

void wykonajRuch(bool dir){
  if (dir){
  switch(krok){
      case 0:
      digitalWrite(IN_1, HIGH);
      digitalWrite(IN_2, LOW);
      digitalWrite(IN_3, LOW);
      digitalWrite(IN_4, LOW);
      break;
    case 1:
      digitalWrite(IN_1, LOW);
      digitalWrite(IN_2, HIGH);
      digitalWrite(IN_3, LOW);
      digitalWrite(IN_4, LOW);
      break;
    case 2:
      digitalWrite(IN_1, LOW);
      digitalWrite(IN_2, LOW);
      digitalWrite(IN_3, HIGH);
      digitalWrite(IN_4, LOW);
      break;
    case 3:
      digitalWrite(IN_1, LOW);
      digitalWrite(IN_2, LOW);
      digitalWrite(IN_3, LOW);
      digitalWrite(IN_4, HIGH);
      break;
      }
  }
  else{
    switch(krok){
     case 0:
        digitalWrite(IN_1, LOW);
        digitalWrite(IN_2, LOW);
        digitalWrite(IN_3, LOW);
        digitalWrite(IN_4, HIGH);
        break;
      case 1:
        digitalWrite(IN_1, LOW);
        digitalWrite(IN_2, LOW);
        digitalWrite(IN_3, HIGH);
        digitalWrite(IN_4, LOW);
        break;
      case 2:
        digitalWrite(IN_1, LOW);
        digitalWrite(IN_2, HIGH);
        digitalWrite(IN_3, LOW);
        digitalWrite(IN_4, LOW);
        break;
      case 3:
        digitalWrite(IN_1, HIGH);
        digitalWrite(IN_2, LOW);
        digitalWrite(IN_3, LOW);
        digitalWrite(IN_4, LOW);
        break;
    }
  }
  krok++;
  if (krok > 3) {
    krok = 0;
  }
}
